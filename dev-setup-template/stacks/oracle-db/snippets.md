# Oracle Database — Patterns & Anti-Patterns

## Table mit Identity + Constraints (DO)
```sql
-- V001__create_customers_table.sql
CREATE TABLE CUSTOMERS (
    CUSTOMER_ID NUMBER(19) GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    EMAIL       VARCHAR2(255) NOT NULL,
    FULL_NAME   VARCHAR2(200) NOT NULL,
    STATUS      VARCHAR2(20) DEFAULT 'ACTIVE' NOT NULL,
    CREATED_AT  TIMESTAMP WITH TIME ZONE DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT UQ_CUSTOMERS_EMAIL UNIQUE (EMAIL),
    CONSTRAINT CK_CUSTOMERS_STATUS CHECK (STATUS IN ('ACTIVE','INACTIVE','BLOCKED'))
);

-- Rollback: DROP TABLE CUSTOMERS;
```

## Foreign Key mit Index (DO)
```sql
-- V002__create_orders_table.sql
CREATE TABLE ORDERS (
    ORDER_ID    NUMBER(19) GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    CUSTOMER_ID NUMBER(19) NOT NULL,
    TOTAL       NUMBER(19,4) NOT NULL,
    CREATED_AT  TIMESTAMP WITH TIME ZONE DEFAULT SYSTIMESTAMP NOT NULL,
    CONSTRAINT FK_ORDERS_CUSTOMER FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMERS(CUSTOMER_ID)
);

CREATE INDEX IDX_ORDERS_CUSTOMER_ID ON ORDERS(CUSTOMER_ID);

-- Rollback: DROP TABLE ORDERS;
```

## Pagination (DO — Oracle 12c+)
```sql
SELECT CUSTOMER_ID, EMAIL, FULL_NAME
FROM CUSTOMERS
WHERE STATUS = 'ACTIVE'
ORDER BY CREATED_AT DESC
OFFSET :offset ROWS FETCH NEXT :pageSize ROWS ONLY;
```

## Anti-Pattern: String Concatenation (DON'T)
```sql
-- FALSCH — SQL Injection Risiko
EXECUTE IMMEDIATE 'SELECT * FROM CUSTOMERS WHERE EMAIL = ''' || v_email || '''';

-- RICHTIG — Bind Variables
EXECUTE IMMEDIATE 'SELECT * FROM CUSTOMERS WHERE EMAIL = :1' USING v_email;
```
